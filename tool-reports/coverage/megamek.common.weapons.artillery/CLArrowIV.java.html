        name = "Arrow IV";

        addLookupName("CLArrowIVSystem");

        heat = 10;

        shortRange = 1; //

        criticals = 12;

            .setClanApproximate(false, false, false,false, false)
            .setPrototypeFactions(F_CHH)
            .setStaticTechLevel(SimpleTechLevel.ADVANCED);
    }

}
